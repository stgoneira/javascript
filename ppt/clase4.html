## [Math functions]() ```javascript Math.abs(p) // devuelve el valor absoluto. // abs(-2) = 2 | abs(2) = 2 | abs(-3) = 3 Math.random() // devuelve número pseudo aleatorio entre 0 y 1 Math.round(p) // devuelve un número redondeado al entero más cercano // round(2.5) = 3 | round(2.8) = 3 | round(2.2) = 2 Math.ceil(p) // aprox. hacia arriba // ceil(1.01) = 2 | ceil(2.5) = 3 Math.floor(p) // aprox. hacia abajo // floor(1.9) = 1 | floor(1.3) = 1 Math.max([x[, y[, …]]]) // retorna el máximo de un conjunto de valores Math.min([x[, y[, …]]]) // retorna el mínimo de un conjunto de valores // min(1, 2, 3, -2) = -2 // max(1, 2, 3, -2) = 3 ``` [Ver todas las funciones](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Math)
## [Funciones Flecha]() ```javascript (param1, param2, …, paramN) => { statements } (param1, param2, …, paramN) => expression // equivalent to: => { return expression; } // Parentheses are optional when there's only one parameter name: (singleParam) => { statements } singleParam => { statements } // The parameter list for a function with no parameters // should be written with a pair of parentheses. () => { statements } ```
## [Funciones Flecha]() ### Ejemplo #1 ```javascript let sumar = (x1, x2) => x1 + x2; let sumar = (x1, x2) => {return x1 + x2}; // lo mismo let sumar = function(x1, x2) {return x1 + x2}; // lo mismo sumar(1, 1); // 2 ```
## [Funciones Flecha]() ### Ejemplo #2 ```javascript let saludar = nombre => { console.log('Hola '+nombre) }; let saludar = (nombre) => { console.log('Hola '+nombre) }; let saludar = function(nombre) { console.log('Hola '+nombre); }; saludar('Juan'); ```
## [Funciones Flecha]() ### Ejemplo #3 ```javascript // devuelve un número aleatorio let rgbRand = () => Math.round(Math.random() * 255); let rgbRand = () => { return Math.round(Math.random() * 255) }; // lo mismo let rgbRand = function(){ return Math.round(Math.random() * 255); }; console.log( rgbRand() ); ```

## [Prevenir comportamiento]() ```html <form> <div> <label for="fname">First name: </label> <input id="fname" type="text"> </div> <div> <label for="lname">Last name: </label> <input id="lname" type="text"> </div> <div> <input id="submit" type="submit"> </div> </form> <p></p> ```
## [Prevenir comportamiento]() ```javascript const form = document.querySelector('form'); const fname = document.getElementById('fname'); const lname = document.getElementById('lname'); const para = document.querySelector('p'); form.onsubmit = function(e) { if (fname.value === '' || lname.value === '') { e.preventDefault(); para.textContent = 'You need to fill in both names!'; } } ```
